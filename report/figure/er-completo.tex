% ------- attributi impilati su un lato, con archi PERPENDICOLARI al bordo -------
% i punti d'attacco sono distribuiti sul bordo (non convergono in un unico punto)

% lato SINISTRO: archi orizzontali, etichette a sinistra
% #1 nodo  #2 offset y del primo  #3 passo  #4 lunghezza arco  #5 lista stile/testo
\newcommand{\stackL}[5]{%
  \foreach[count=\i from 0] \s/\t in {#5}{%
    \coordinate (p) at ($(#1.west)+(0,{#2-\i*#3})$);
    \node[\s] (#1-l\i) at ($(p)+(-#4,0)$) {};
    \node[et,left] at ($(#1-l\i)+(-0.12,0)$) {\t};
    \draw[link] (p) -- (#1-l\i);
  }%
}
% lato DESTRO: archi orizzontali, etichette a destra
\newcommand{\stackR}[5]{%
  \foreach[count=\i from 0] \s/\t in {#5}{%
    \coordinate (p) at ($(#1.east)+(0,{#2-\i*#3})$);
    \node[\s] (#1-r\i) at ($(p)+(#4,0)$) {};
    \node[et,right] at ($(#1-r\i)+(0.12,0)$) {\t};
    \draw[link] (p) -- (#1-r\i);
  }%
}
% comb in ALTO: archi verticali, etichette in alto, altezze ALTERNATE (vicino/lontano)
% #1 nodo  #2 offset x del primo  #3 passo  #4 arco corto  #5 arco lungo  #6 lista
\newcommand{\combT}[6]{%
  \foreach[count=\i from 0] \s/\t in {#6}{%
    \coordinate (p) at ($(#1.north)+({#2+\i*#3},0)$);
    \pgfmathparse{Mod(\i,2)==0 ? #4 : #5}\let\hy\pgfmathresult
    \node[\s] (#1-t\i) at ($(p)+(0,\hy)$) {};
    \node[et,above] at ($(#1-t\i)+(0,0.1)$) {\t};
    \draw[link] (p) -- (#1-t\i);
  }%
}
% comb in BASSO: archi verticali, etichette in basso, profondita ALTERNATE
\newcommand{\combB}[6]{%
  \foreach[count=\i from 0] \s/\t in {#6}{%
    \coordinate (p) at ($(#1.south)+({#2+\i*#3},0)$);
    \pgfmathparse{Mod(\i,2)==0 ? #4 : #5}\let\hy\pgfmathresult
    \node[\s] (#1-b\i) at ($(p)+(0,-\hy)$) {};
    \node[et,below] at ($(#1-b\i)+(0,-0.1)$) {\t};
    \draw[link] (p) -- (#1-b\i);
  }%
}
